1. What is a Dictionary?

A dictionary is an ordered (Python 3.7+), mutable collection of key → value pairs. Keys must be unique and hashable; values can be any Python object.

"Alice"

PropertyDescription
OrderedPreserves insertion order (CPython 3.6+, guaranteed 3.7+)
MutableKeys/values can be added, updated, or removed
Unique keysDuplicate keys are overwritten; values can repeat
Hashable keys onlyKeys must be immutable: str, int, float, tuple (no list/dict/set)
Any value typeValues can be anything — lists, functions, other dicts…
O(1) averageLookup, insert, delete are constant time on average

2. Creating Dictionaries

keys = ["x", "y", "z"]
d4 = dict.fromkeys(keys)          # {'x': None, 'y': None, 'z': None}
d5 = dict.fromkeys(keys, 0)       # {'x': 0, 'y': 0, 'z': 0}

empty = {}
empty = dict()

# ── Comprehension ──────────────────────────────────────────

# ── Mixed value types ──────────────────────────────────────
config = {
    "host":    "localhost",
    "port":    8080,
    "debug":   True,
    "tags":    ["web", "api"],
    "options": {"timeout": 30},
}
💡
dict.fromkeys(keys, value) warning: The default value is shared across all keys. If the value is mutable (e.g., a list), all keys will reference the same list object. Use a comprehension instead: {k: [] for k in keys}.

3. Accessing Values

Bracket Notation []

d = {"a": 1, "b": 2}
d["a"]      # 1
d["z"]      # ❌ KeyError

get() — safe access

d.get("a")          # 1
d.get("z")          # None  (no error)
d.get("z", "N/A")   # "N/A" (custom default)
# All keys, values, items
d = {"x": 10, "y": 20, "z": 30}
d.keys()     # dict_keys(['x', 'y', 'z'])
d.values()   # dict_values([10, 20, 30])
d.items()    # dict_items([('x',10),('y',20),('z',30)])

# Convert to lists
list(d.keys())    # ['x', 'y', 'z']
list(d.values())  # [10, 20, 30]

# Nested access
config = {"db": {"host": "localhost", "port": 5432}}
config["db"]["host"]             # "localhost"
config.get("db", {}).get("port")  # 5432 — safe nested access
✏️

4. Adding & Updating

d = {"a": 1}

# Add a new key
d["b"] = 2          # {'a': 1, 'b': 2}

# Update existing key
d["a"] = 99         # {'a': 99, 'b': 2}

# update() — merge another dict or iterable of pairs
d.update({"c": 3, "d": 4})        # adds/updates c and d
d.update(e=5, f=6)               # keyword args
d.update([("g", 7), ("h", 8)])   # list of tuples

# setdefault() — add key ONLY if it doesn't exist
d.setdefault("a", 0)   # returns 99 — key exists, no change
d.setdefault("z", 0)   # inserts z=0, returns 0

# Python 3.9+ — merge operator |=
d |= {"x": 100}       # in-place update

6. Checking & Testing

d = {"name": "Alice", "age": 30}

# Key membership — O(1)
"name" in d             # True
"email" not in d        # True

# Value membership — O(n)
"Alice" in d.values()   # True

not d                    # False (non-empty dict is truthy)
not {}                   # True  (empty dict is falsy)

# Type check

7. Iterating Over Dictionaries

d = {"a": 1, "b": 2, "c": 3}

# ── Iterate keys (default) ────────────────────────────
for k in d:
    print(k, d[k])

# ── Iterate keys explicitly ───────────────────────────
for k in d.keys():
    print(k)

# ── Iterate values ────────────────────────────────────
for v in d.values():
    print(v)

# ── Iterate key-value pairs (most common) ─────────────
for k, v in d.items():
    print(f"{k} → {v}")

# ── Sorted iteration ──────────────────────────────────
for k in sorted(d):
    print(k, d[k])

for k, v in sorted(d.items(), key=lambda x: x[1]):   # sort by value
    print(k, v)

# ── Reversed iteration (Python 3.8+) ──────────────────
for k in reversed(d):
    print(k)

# ── Enumerate with index ──────────────────────────────
for i, (k, v) in enumerate(d.items()):
    print(i, k, v)
⚠️
Do not add or remove keys while iterating over a dict — this raises RuntimeError: dictionary changed size during iteration. Iterate over list(d.items()) or collect keys to delete first.

9. Dictionary Comprehensions

Syntax: { key_expr: val_expr for item in iterable [if condition] }

# Basic
squares = {x: x**2 for x in range(6)}
# {0: 0, 1: 1, 2: 4, 3: 9, 4: 16, 5: 25}

# With condition
even_sq = {x: x**2 for x in range(10) if x % 2 == 0}
# {0: 0, 2: 4, 4: 16, 6: 36, 8: 64}

# Invert a dict (swap keys and values)
d      = {"a": 1, "b": 2, "c": 3}
inv    = {v: k for k, v in d.items()}   # {1:'a', 2:'b', 3:'c'}

# Filter keys
scores = {"Alice": 88, "Bob": 55, "Carol": 91}
passed = {k: v for k, v in scores.items() if v >= 60}
# {'Alice': 88, 'Carol': 91}

# Transform values
upper = {k: v.upper() for k, v in {"a": "hello", "b": "world"}.items()}

# From two lists
keys   = ["name", "age", "city"]
values = ["Alice", 30, "Paris"]
d = {k: v for k, v in zip(keys, values)}

# Nested comprehension — flatten nested dict
nested = {"a": {"x": 1}, "b": {"y": 2}}
flat   = {f"{ok}.{ik}": iv for ok, ov in nested.items() for ik, iv in ov.items()}
# {'a.x': 1, 'b.y': 2}
🔀

10. Merging Dictionaries

a = {"x": 1, "y": 2}
b = {"y": 99, "z": 3}   # b's value for 'y' wins on conflict

# ── Python 3.9+ — merge operator | ──────────────────────
merged = a | b           # {'x':1, 'y':99, 'z':3} — new dict
a |= b                   # in-place update a

# ── Unpacking (Python 3.5+) ──────────────────────────────
merged = {**a, **b}      # {'x':1, 'y':99, 'z':3}
merged = {**a, **b, "extra": 42}   # add extra keys inline

# ── update() — mutates a ─────────────────────────────────
a.update(b)              # a modified in place

# ── ChainMap — non-destructive layering ──────────────────
from collections import ChainMap
cm = ChainMap(b, a)      # looks up b first, then a
cm["y"]                  # 99   (from b)
cm["x"]                  # 1    (from a)

# ── Merge preserving all values (list per key) ───────────
from collections import defaultdict
merged = defaultdict(list)
for d in [a, b]:
    for k, v in d.items():
        merged[k].append(v)
# {'x': [1], 'y': [2, 99], 'z': [3]}
💡
When two dicts share a key and you merge, the right-hand dict wins for both | and {**a, **b}. Earlier keys in ChainMap win (first-write-wins lookup).
📦

11. Nested Dictionaries

org = {
    "CEO": {
        "name": "Alice",
        "reports": {
            "CTO": {"name": "Bob", "reports": {}},
            "CFO": {"name": "Carol", "reports": {}},
        }
    }
}

# Access nested value
org["CEO"]["reports"]["CTO"]["name"]   # "Bob"

# Safe deep access with .get() chaining
org.get("CEO", {}).get("reports", {}).\
    get("CTO", {}).get("name")          # "Bob"

# Generic deep_get helper
def deep_get(d, *keys, default=None):
    for k in keys:
        if not isinstance(d, dict):
            return default
        d = d.get(k, default)
    return d

deep_get(org, "CEO", "reports", "CTO", "name")  # "Bob"
deep_get(org, "CEO", "reports", "COO", "name")  # None

# Update nested value
org["CEO"]["reports"]["CTO"]["name"] = "Robert"

# Copy (shallow vs deep)
import copy
shallow = org.copy()            # inner dicts still shared
deep    = copy.deepcopy(org)    # fully independent copy

12. Specialized Dictionary Types

defaultdict — automatic default values

from collections import defaultdict

# Grouping
items = [("a", 1), ("b", 2), ("a", 3), ("b", 4)]
grouped = defaultdict(list)
for k, v in items:
    grouped[k].append(v)
# defaultdict(list, {'a': [1, 3], 'b': [2, 4]})

# Word frequency count
freq = defaultdict(int)
for word in "the cat sat on the mat".split():
    freq[word] += 1
# {'the': 2, 'cat': 1, 'sat': 1, 'on': 1, 'mat': 1}

# Nested defaultdict
nested = defaultdict(lambda: defaultdict(int))
nested["A"]["x"] += 1    # no KeyError

Counter — frequency counting

from collections import Counter

c = Counter("abracadabra")
# Counter({'a': 5, 'b': 2, 'r': 2, 'c': 1, 'd': 1})

c = Counter(["apple", "banana", "apple", "cherry"])
c.most_common(2)        # [('apple', 2), ('banana', 1)]

c["apple"]               # 2
c["missing"]             # 0 — no KeyError!

# Counter arithmetic
c1 = Counter(a=3, b=1)
c2 = Counter(a=1, b=2)
c1 + c2   # Counter({'a': 4, 'b': 3})
c1 - c2   # Counter({'a': 2})        — negatives dropped
c1 & c2   # Counter({'a': 1, 'b': 1}) — min of each
c1 | c2   # Counter({'a': 3, 'b': 2}) — max of each

OrderedDict — explicit ordering (legacy)

from collections import OrderedDict

od = OrderedDict()
od["first"]  = 1
od["second"] = 2
od["third"]  = 3

od.move_to_end("first")       # move to end
od.move_to_end("third", last=False)  # move to front

# Equality — OrderedDict considers order, dict doesn't
OrderedDict([("a",1),("b",2)]) == OrderedDict([("b",2),("a",1)])
# False

ChainMap — layered lookups

from collections import ChainMap

defaults = {"color": "red", "user": "guest"}
env      = {"user": "admin"}          # overrides default
cli      = {"color": "blue"}          # highest priority

config = ChainMap(cli, env, defaults)  # left-to-right priority
config["color"]   # "blue"  (from cli)
config["user"]    # "admin" (from env)

# Writes go to the FIRST map
config["new_key"] = "hi"    # added to cli

Types.MappingProxyType — read-only dict

from types import MappingProxyType

d     = {"a": 1, "b": 2}
proxy = MappingProxyType(d)

proxy["a"]        # 1 — read OK
proxy["c"] = 3   # ❌ TypeError: 'mappingproxy' does not support item assignment

13. Type Annotations

# ── Python 3.9+ — built-in generics ──────────────────────
def get_score(scores: dict[str, int], name: str) -> int:
    return scores.get(name, 0)

mapping: dict[str, list[int]] = {"a": [1, 2], "b": [3, 4]}

# ── Python 3.8 and earlier — typing module ───────────────
from typing import Dict, List, Optional, Any

scores: Dict[str, int]              = {"Alice": 95}
config: Dict[str, Any]             = {"port": 8080, "debug": True}
nested: Dict[str, Dict[str, int]]  = {"a": {"x": 1}}

# ── TypedDict — typed dict with fixed keys ────────────────
from typing import TypedDict

class Person(TypedDict):
    name: str
    age:  int
    city: str

class PersonOpt(TypedDict, total=False):  # all keys optional
    name: str
    age:  int

alice: Person = {"name": "Alice", "age": 30, "city": "Paris"}

# ── Mapping — read-only protocol ──────────────────────────
from collections.abc import Mapping

def process(data: Mapping[str, int]) -> int:
    return sum(data.values())

16. Common Patterns & Recipes

Frequency / Word Count

words = ["the", "cat", "sat", "on", "the", "mat"]

# Method 1 — get() with default
freq = {}
for w in words:
    freq[w] = freq.get(w, 0) + 1

# Method 2 — defaultdict
from collections import defaultdict
freq = defaultdict(int)
for w in words: freq[w] += 1

# Method 3 — Counter (best)
from collections import Counter
freq = Counter(words)

Grouping Items

people = [
    {"name": "Alice", "dept": "Eng"},
    {"name": "Bob",   "dept": "HR"},
    {"name": "Carol", "dept": "Eng"},
]
by_dept = defaultdict(list)
for p in people:
    by_dept[p["dept"]].append(p["name"])
# {'Eng': ['Alice', 'Carol'], 'HR': ['Bob']}

Invert a Dictionary

# Simple invert (unique values)
d   = {"a": 1, "b": 2, "c": 3}
inv = {v: k for k, v in d.items()}   # {1:'a', 2:'b', 3:'c'}

# Invert with duplicate values — collect keys in list
d   = {"a": 1, "b": 2, "c": 1}
inv = defaultdict(list)
for k, v in d.items(): inv[v].append(k)
# {1: ['a', 'c'], 2: ['b']}

Sort a Dict by Value

scores = {"Alice": 88, "Bob": 55, "Carol": 91}

# Ascending
asc  = dict(sorted(scores.items(), key=lambda x: x[1]))
# Descending
desc = dict(sorted(scores.items(), key=lambda x: x[1], reverse=True))

# Top N items
top2 = dict(Counter(scores).most_common(2))

Caching / Memoisation

cache = {}

def fib(n):
    if n in cache:
        return cache[n]
    if n <= 1:
        return n
    cache[n] = fib(n-1) + fib(n-2)
    return cache[n]

# Better: use functools.lru_cache
from functools import lru_cache
@lru_cache(maxsize=None)
def fib(n):
    return n if n <= 1 else fib(n-1) + fib(n-2)

Dispatch Table (replace if/elif chains)

def add(a, b): return a + b
def sub(a, b): return a - b
def mul(a, b): return a * b

ops = {"+": add, "-": sub, "*": mul}

result = ops.get("+", lambda a, b: None)(3, 4)   # 7

Structural Pattern Matching (Python 3.10+)

command = {"action": "move", "direction": "north", "steps": 3}

match command:
    case {"action": "quit"}:
        print("Quitting")
    case {"action": "move", "direction": d, "steps": n}:
        print(f"Move {n} steps {d}")    # Move 3 steps north
    case _:
        print("Unknown command")

17. Common Pitfalls

⚠️
Pitfall 1 — KeyError on missing key
d["missing"] raises KeyError. Use d.get("missing", default) for safe access.
⚠️
Pitfall 2 — Modifying a dict while iterating
for k in d: del d[k] → RuntimeError.
Fix: for k in list(d): del d[k] or collect keys then delete.
⚠️
Pitfall 3 — Unhashable key types
{[1, 2]: "value"} → TypeError: unhashable type: 'list'.
Use tuples as keys: {(1, 2): "value"}.
⚠️
Pitfall 4 — Shallow copy shares nested objects
d2 = d.copy(): top-level is copied but nested dicts/lists are still shared.
Use import copy; d2 = copy.deepcopy(d) for full independence.
⚠️
Pitfall 5 — fromkeys with mutable default
dict.fromkeys(["a","b"], []) — all keys reference the same list!
Fix: {k: [] for k in ["a","b"]}.
⚠️
Pitfall 6 — Comparing dicts with is
d1 is d2 checks identity (same object), not equality.
Use d1 == d2 to compare contents.
📋

19. Cheatsheet

quick reference
# ── Create ─────────────────────────────────────────────────
d = {"a": 1, "b": 2}                    # literal
d = dict(a=1, b=2)                     # keyword args
d = dict.fromkeys(["x","y"], 0)       # all same value
d = {k: v for k, v in pairs}           # comprehension

# ── Access ─────────────────────────────────────────────────
d["a"]                                  # KeyError if missing
d.get("a", "default")                  # safe — no error
d.keys()  /  d.values()  /  d.items()  # views

# ── Modify ─────────────────────────────────────────────────
d["c"] = 3                              # add / update
d.update({"d": 4, "e": 5})             # bulk update
d.setdefault("f", []).append(1)        # insert if absent

# ── Remove ─────────────────────────────────────────────────
del d["a"]                              # KeyError if missing
d.pop("b", None)                        # safe remove
d.popitem()                             # LIFO (k, v) pair
d.clear()                               # empty

# ── Check ──────────────────────────────────────────────────
"k" in d                               # O(1) key check
len(d)                                  # count of pairs
bool(d)                                 # False if empty

# ── Iterate ────────────────────────────────────────────────
for k, v in d.items(): ...              # most common
sorted(d.items(), key=lambda x: x[1])  # sort by value

# ── Merge ──────────────────────────────────────────────────
a | b                                   # new dict (3.9+)
{**a, **b}                              # new dict (3.5+)
a |= b ; a.update(b)                   # in-place

# ── Copy ───────────────────────────────────────────────────
d.copy()                                # shallow
copy.deepcopy(d)                        # deep

# ── Special types ──────────────────────────────────────────
defaultdict(list)                       # auto default
Counter(iterable)                       # frequency count
ChainMap(d1, d2)                        # layered lookup
MappingProxyType(d)                     # read-only view

# ── Recipes ────────────────────────────────────────────────
{v: k for k, v in d.items()}           # invert dict
{k: v for k, v in d.items() if v > 0}  # filter by value

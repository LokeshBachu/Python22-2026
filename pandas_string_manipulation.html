How .str works — dtype requirements and NaN propagation

Core concept
import pandas as pd
import numpy as np
import re

# ── .str is available on object, string, and category dtype ───────
s = pd.Series(["  Alice Johnson  ", "bob SMITH", "CAROL white", np.nan, "  dave Brown"])

print("dtype:", s.dtype)           # object
print("lower:", s.str.lower().tolist())
print("strip:", s.str.strip().tolist())

# ── NaN rows propagate as NaN by default ─────────────────────────
print("\nNaN propagation:")
print(s.str.upper())

# ── na= fills NaN with a replacement value ───────────────────────
print("\nWith na='' fill:")
print(s.str.upper().fillna(""))     # or use na= in methods that support it
print(s.str.contains("alice", case=False, na=False))   # na=False: NaN → False

# ── .str on non-object dtype raises AttributeError ───────────────
n = pd.Series([1, 2, 3])
try:
    n.str.upper()
except AttributeError as e:
    print(f"\nAttributeError: {e}")

# ── Works on category dtype too ───────────────────────────────────
cat = pd.Series(["shipped", "pending", "shipped"], dtype="category")
print("\ncategory .str.upper():", cat.str.upper().tolist())

# ── Chain multiple .str calls ─────────────────────────────────────
cleaned = s.str.strip().str.lower().str.replace(r"\s+", " ", regex=True)
print("\nChained:", cleaned.tolist())
Output

strip: ['Alice Johnson', 'bob SMITH', 'CAROL white', nan, 'dave Brown']

1        BOB SMITH

Case Conversion

All case methods

Case
s = pd.Series(["hello WORLD", "the Quick Brown FOX", "ALICE johnson", np.nan])

# ── The six case methods ─────────────────────────────────────────
print("lower()     :", s.str.lower().tolist())
print("upper()     :", s.str.upper().tolist())
print("title()     :", s.str.title().tolist())    # Title Case Each Word
print("capitalize():", s.str.capitalize().tolist()) # First letter only
print("swapcase()  :", s.str.swapcase().tolist())  # Flip each letter
print("casefold()  :", s.str.casefold().tolist())  # Aggressive lower (ß→ss)

# ── casefold vs lower — German ß example ─────────────────────────
german = pd.Series(["Straße", "STRASSE"])
print("\nlower():   ", german.str.lower().tolist())     # ['straße', 'strasse']
print("casefold():", german.str.casefold().tolist())  # ['strasse', 'strasse'] ← same!

df2 = df.copy()

df_messy = pd.DataFrame(columns=[

    "Phone Number", "OrderID", "  ShipDate  "
])
df_messy.columns = (
    df_messy.columns
    .str.strip()
    .str.lower()
    .str.replace(r"[^\w\s]", "", regex=True)

)
print("\nSnake_case columns:", df_messy.columns.tolist())

Search & Test Methods

contains / startswith / endswith / match / fullmatch / find

Search
s = pd.Series(["alice@example.com", "bob.smith@gmail.com",
               "not-an-email", "carol@yahoo.co.uk", np.nan])

# ── contains: substring or regex search ──────────────────────────
print("contains 'gmail':", s.str.contains("gmail", na=False).tolist())
print("contains regex :", s.str.contains(r"@\w+\.\w+", regex=True, na=False).tolist())

# ── case-insensitive contains ─────────────────────────────────────
print("contains (no case):", s.str.contains("GMAIL", case=False, na=False).tolist())

# ── startswith / endswith ─────────────────────────────────────────
print("\nstartswith 'alice':", s.str.startswith("alice", na=False).tolist())
print("endswith   '.com' :", s.str.endswith(".com", na=False).tolist())

# Multiple prefixes/suffixes (pass a tuple)
print("startswith ('alice','carol'):",
      s.str.startswith(("alice", "carol"), na=False).tolist())

# ── match: test against regex at the START of string ─────────────
# match == re.match (anchored at start)  ←→  contains == re.search
codes = pd.Series(["ORD-001", "XORD-002", "ORD-ABC", "ORD-123"])
print("\nmatch '^ORD-\\d+':", codes.str.match(r"^ORD-\d+").tolist())

# ── fullmatch: entire string must match ──────────────────────────
print("fullmatch '^ORD-\\d{3}$':", codes.str.fullmatch(r"^ORD-\d{3}$").tolist())

# ── find / rfind: position of substring (-1 if not found) ────────
text = pd.Series(["hello world", "world peace", "no match here"])
print("\nfind('world')  :", text.str.find("world").tolist())    # [6, 0, -1]
print("index('world') :")                                        # raises on -1

# ── count: occurrences of a pattern ──────────────────────────────
print("\ncount('o'):", text.str.count("o").tolist())

# ── Filter DataFrame rows using string tests ─────────────────────
gmail_users = df[df["email"].str.endswith("gmail.com", na=False)]
python_devs = df[df["tags"].str.contains("python", na=False)]
print("\nGmail users:", gmail_users["email"].tolist())
print("Python devs:", len(python_devs))
Output
contains 'gmail': [False, True, False, False, False]
contains regex  : [True, True, False, True, False]   ← "not-an-email" fails ✅

startswith ('alice','carol'): [True, False, False, True, False]

match '^ORD-\d+':    [True, False, True, True]   ← XORD fails, ORD-ABC passes
fullmatch '^ORD-\d{3}$': [True, False, False, True]  ← strict: must be exactly 3 digits

find('world'): [6, 0, -1]   ← -1 means not found

Gmail users: ['bob.smith@gmail.com']
Python devs: 3
match vs contains vs fullmatch: match = anchored at start (like re.match). contains = anywhere in string (like re.search). fullmatch = entire string must match (like re.fullmatch). Use fullmatch for strict validation.

Replace & Substitute

str.replace() — literal and regex

Replace
# ── literal replace ──────────────────────────────────────────────
s = pd.Series(["hello world", "world is round", "hello hello"])
print(s.str.replace("hello", "hi", regex=False).tolist())
# → ['hi world', 'world is round', 'hi hi']  (all occurrences)

# ── n= limits replacement count ─────────────────────────────────
print(s.str.replace("hello", "hi", n=1, regex=False).tolist())
# → ['hi world', 'world is round', 'hi hello']  (max 1 replacement)

# ── regex replace ────────────────────────────────────────────────
noisy = pd.Series(["$1,234.56", "€ 2,500.00", "USD 890", "FREE"])

# Remove everything except digits and dot
clean_num = noisy.str.replace(r"[^\d.]", "", regex=True)
print("\nStrip non-numeric:", clean_num.tolist())

# Replace multiple whitespace with single space
messy_spaces = pd.Series(["hello   world", "too    many    spaces"])
print("Normalise spaces:", messy_spaces.str.replace(r"\s+", " ", regex=True).str.strip().tolist())

# ── Backreference in replacement ─────────────────────────────────
phones = pd.Series(["5551234567", "5559876543"])
# Format: (555) 123-4567
formatted = phones.str.replace(
    r"(\d{3})(\d{3})(\d{4})",
    r"(\1) \2-\3",
    regex=True
)
print("\nFormatted phones:", formatted.tolist())

# ── Replace using a callable (function) ──────────────────────────
words = pd.Series(["foo 123 bar 456", "abc 7 def 89"])
# Double every digit sequence found
doubled = words.str.replace(
    r"\d+",
    lambda m: str(int(m.group()) * 2),
    regex=True
)
print("Doubled digits:", doubled.tolist())

# ── Strip currency symbols — real-world pattern ───────────────────
df2 = df.copy()
df2["amount_clean"] = (
    df2["amount"]
    .str.replace(r"[$,€£]", "", regex=True)
    .str.strip()
    .replace("FREE", "0")
)
df2["amount_num"] = pd.to_numeric(df2["amount_clean"], errors="coerce")
print("\nAmount values:", df2["amount_num"].tolist())
Output

Normalise spaces : ['hello world', 'too many spaces']
Formatted phones : ['(555) 123-4567', '(555) 987-6543']  ✅
Doubled digits   : ['foo 246 bar 912', 'abc 14 def 178']

Amount values: [1234.56, 89.0, 2500.0, 450.75, 0.0, 3100.0]  ✅

translate() — character-level replacement (fastest)

# str.translate() uses a character translation table — much faster
# than regex when replacing individual characters

# Remove punctuation from text
import string

PUNCT_TABLE = str.maketrans("", "", string.punctuation)

# Replace specific characters (e.g. normalize dashes)

import unicodedata

# ── Strip accents (NFD decompose → drop combining chars) ─────────
def strip_accents(series: pd.Series) -> pd.Series:
    """Remove diacritic marks: é→e, ü→u, ñ→n, etc."""
    return series.apply(
        lambda x: (
            unicodedata.normalize("NFD", x)
            .encode("ascii", "ignore")
            .decode("ascii")
        ) if pd.notna(x) else x
    )

str.partition(sep) always splits into exactly 3 parts: (before, sep, after). Unlike split(), it never returns variable-length results and always succeeds, making it safer for structured strings.

Advanced Regex Patterns

Lookaheads, lookbehinds, non-capturing groups, named groups, verbose patterns, and compiled-regex strategies for complex extraction tasks.

Lookahead and lookbehind assertions

Advanced
# ── Positive lookahead (?=...) — match X only when followed by Y ──
s = pd.Series(["100px", "200em", "50px", "75", "30%"])

px_only = s.str.extract(r"(\d+)(?=px)")[0]
print("Number before 'px' only:", px_only.tolist())
# ['100', NaN, '50', NaN, NaN]  ← 200em and bare 75 don't match

# ── Positive lookbehind (?<=...) — match X only when preceded by Y ─
prices = pd.Series(["$100", "€200", "£50", "100"])

dollar_amt = prices.str.extract(r"(?<=\$)(\d+)")[0]
print("\nDollar amounts only:", dollar_amt.tolist())
# ['100', NaN, NaN, NaN]  ← €200 and bare 100 excluded

# ── Negative lookbehind (?<!...) — match X NOT preceded by Y ──────
no_prefix = prices.str.extract(r"(?<![$€£])(\d+)")[0]
print("No currency prefix:", no_prefix.tolist())
# [NaN, NaN, NaN, '100']  ← only bare 100 matches

# ── Negative lookahead (?!...) — match X NOT followed by Y ────────
numbers = pd.Series(["100px", "200em", "300"])
not_px = numbers.str.extract(r"(\d+)(?!px)")[0]
print("\nNot followed by px:", not_px.tolist())

# ── Named groups + lookarounds: extract tag contents ─────────────
html = pd.Series(["<b>Hello</b>", "<i>World</i>", "<span>Text</span>"])
content = html.str.extract(r"(?<=>)(?P<text>[^<]+)(?=</)")
print("\nHTML tag content:", content["text"].tolist())
# ['Hello', 'World', 'Text']  ← no tag name in result, only text content
Lookaround summary
Syntax       Name                    Meaning
(?=Y)        Positive lookahead      X only if FOLLOWED BY Y       (Y not consumed)
(?!Y)        Negative lookahead      X only if NOT followed by Y
(?<=Y)       Positive lookbehind     X only if PRECEDED BY Y       (Y not consumed)
(?<!Y)       Negative lookbehind     X only if NOT preceded by Y

Zero-width: lookarounds check context but do NOT include the
context text in the match / extract / replace output.

Non-capturing groups, alternation, and re.VERBOSE

import re

# ── Non-capturing group (?:...) — group without capturing ────────
# Every (...) in a pattern adds a column to str.extract() result
# Use (?:...) when you need grouping but NOT the group in the output
logs = pd.Series(["ERROR: disk full", "WARN: low memory",
                  "INFO: started", "CRITICAL: db down"])

# BAD: captures 2 groups → 2 columns (level AND message)
bad = logs.str.extract(r"(ERROR|CRITICAL): (.+)")
print("Capturing both groups (2 cols):")
print(bad)

# GOOD: non-capturing for level → only message column
good = logs.str.extract(r"(?:ERROR|CRITICAL): (.+)")[0]
print("\nNon-capturing level (1 col):", good.dropna().tolist())

# ── re.VERBOSE: annotated regex for complex patterns ─────────────
DATE_PAT = re.compile(r"""
    (?P<year>  \d{4})   # 4-digit year
    [-/]                 # separator: dash or slash
    (?P<month> \d{1,2}) # 1 or 2 digit month
    [-/]                 # separator
    (?P<day>   \d{1,2}) # 1 or 2 digit day
""", re.VERBOSE)

dates = pd.Series(["2024-01-15", "2023/12/31", "not-a-date", "2025-3-7"])
parsed = dates.str.extract(DATE_PAT)
print("\nVerbose date regex:")
print(parsed)

# ── Compiled pattern reuse in str accessor methods ────────────────
# pandas .str methods accept compiled re.Pattern objects directly

removeprefix() and removesuffix()

Python 3.9+ str.removeprefix() / str.removesuffix() remove a known literal prefix or suffix. They are safer than lstrip()/rstrip() because they operate on a substring, not a character set. Available on the pandas .str accessor.

The lstrip bug vs removeprefix — why it matters

Python 3.9+
s = pd.Series(["https://example.com", "https://other.org", "http://plain.com"])

# ── WRONG: lstrip removes a CHARACTER SET, not a substring ───────
# lstrip('https://') strips any of: h, t, p, s, :, / from the left
bad = s.str.lstrip("https://")
print("lstrip (WRONG)     :", bad.tolist())
# ['example.com', 'other.org', 'plain.com']
# ↑ http://plain.com → 'plain.com'  (http:// accidentally stripped!)

# ── CORRECT: removeprefix removes the exact substring ────────────
good = s.str.removeprefix("https://")
print("removeprefix (✅)  :", good.tolist())
# ['example.com', 'other.org', 'http://plain.com']
# ↑ http://plain.com unchanged — prefix doesn't match exactly

# ── removesuffix: strip a known file extension ───────────────────
files = pd.Series(["report.pdf", "data.csv", "archive.csv", "image.png"])
cleaned = files.str.removesuffix(".csv")
print("\nremovesuffix('.csv'):", cleaned.tolist())
# ['report.pdf', 'data', 'archive', 'image.png']  ← .pdf and .png untouched

# ── Strip environment prefix from config keys ────────────────────
env_keys = pd.Series(["APP_DB_HOST", "APP_DB_PORT", "APP_SECRET", "OTHER_KEY"])
print("\nremoveprefix('APP_'):", env_keys.str.removeprefix("APP_").tolist())
# ['DB_HOST', 'DB_PORT', 'SECRET', 'OTHER_KEY']  ← OTHER_KEY unchanged

# ── Chain removeprefix + removesuffix ────────────────────────────
routes = pd.Series(["/api/v1/users/", "/api/v1/orders/", "/health"])
cleaned_routes = routes.str.removeprefix("/api/v1/").str.removesuffix("/")
print("\nCleaned routes:", cleaned_routes.tolist())
# ['users', 'orders', '/health']  ← '/health' unaffected by prefix strip

# ── Python < 3.9 compatibility alternative ───────────────────────
import re
prefix = "https://"
compat = s.str.replace(r"^" + re.escape(prefix), "", regex=True)
print("\ncompat regex:", compat.tolist())
# Same result as removeprefix — re.escape() ensures special chars are literal

The classic lstrip bug: lstrip('https://') does NOT strip the prefix "https://". It strips the set of characters {'h','t','p','s',':','/'} from the left, one by one, for as long as the next character is in that set. Always use removeprefix() (Python 3.9+) or a regex with ^ anchor for prefix removal.

Batch string cleaning — efficient production pattern

# ── Compile all patterns once at module level ─────────────────────
_RE_SPACES  = re.compile(r"\s+")
_RE_NONNUM  = re.compile(r"[^\d.]")
_RE_PHONE   = re.compile(r"[^\d]")
_RE_EMAIL   = re.compile(r"^[a-z0-9._%+\-]+@[a-z0-9.\-]+\.[a-z]{2,}$")

def clean_string_columns(df: pd.DataFrame) -> pd.DataFrame:
    """Apply all string cleaning in a single pass."""
    df = df.copy()

    # 1. All object columns: strip + collapse spaces
    obj_cols = df.select_dtypes("object").columns
    for col in obj_cols:
        df[col] = (df[col]
                   .str.strip()
                   .str.replace(r"\s+", " ", regex=True))

    # 2. Name → title case
    if "name" in df.columns:
        df["name"] = df["name"].str.title()

    # 3. Email → lowercase + validate flag
    if "email" in df.columns:
        df["email"] = df["email"].str.lower()
        df["email_valid"] = df["email"].str.fullmatch(
            r"^[a-z0-9._%+\-]+@[a-z0-9.\-]+\.[a-z]{2,}$", na=False
        )

    # 4. Phone → digits only
    if "phone" in df.columns:
        df["phone_digits"] = df["phone"].str.replace(r"[^\d]", "", regex=True)

    # 5. Amount → numeric
    if "amount" in df.columns:
        df["amount_num"] = pd.to_numeric(
            df["amount"].str.replace(r"[$,€]", "", regex=True)
                         .replace("FREE", "0"),
            errors="coerce"
        )

    return df

result = clean_string_columns(df)
print(result[["name", "email", "email_valid", "phone_digits", "amount_num"]].to_string())

⚠️ Always use regex=False for literal strings:
str.replace("$29.99", "", regex=False) — the $ is a literal dollar sign. Without regex=False it's a regex end-anchor and won't match. In pandas 2.x this defaults to True with a warning.
⚠️ astype(bool) on strings is wrong:
Any non-empty string is truthy. "False".astype(bool) == True. Use .map({"True": True, "False": False}) or .str.lower().isin({"true","yes","1"}).
💡 na= parameter prevents filtering bugs:
str.contains("foo", na=False) returns False for NaN rows instead of NaN. Without it, boolean indexing like df[mask] raises when mask contains NaN.
✅ Compile regex once for apply loops:
If using apply() with regex, compile the pattern at module level: PAT = re.compile(r"\d+"). Reusing a compiled pattern is ~3× faster than re-compiling on every row.
💡 str.extract() returns a DataFrame:
Even with a single capture group, str.extract() returns a one-column DataFrame, not a Series. Use str.extract(r"(pat)")[0] or add a named group (?P<name>pat) to get a named column.
💡 Use casefold() not lower() for comparisons:
casefold() handles Unicode edge cases like German ß → ss. For standard ASCII data lower() is fine and slightly faster. For multilingual data always use casefold().

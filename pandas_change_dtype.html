How pandas assigns default dtypes on read

import pandas as pd
import numpy as np

    "amount":     ["$29.99","$49.00","$15.50","$89.00","$34.00"],   # should be float

    "is_paid":    ["True","False","True","True","False"],          # should be bool
    "status":     ["shipped","pending","shipped","cancelled","pending"],
    "rating":     ["4", "5", np.nan, "3", "5"],                   # int with NaN!
    "duration_s": ["3600", "7200", "1800", "5400", "900"],        # should be timedelta
})

# What pandas infers by default

print(f"\nMemory before: {raw.memory_usage(deep=True).sum() / 1024:.1f} KB")

Why everything is object: When pandas reads a CSV with mixed or ambiguous data, it defaults to object (Python heap string). This is 5-10× more memory than the correct numeric type and much slower for arithmetic.

# ── Detailed dtype audit DataFrame ──────────────────────────────
def dtype_audit(df: pd.DataFrame) -> pd.DataFrame:
    rows = []
    for col in df.columns:
        s = df[col]
        mem = s.memory_usage(deep=True)
        rows.append({
            "column":      col,
            "dtype":       s.dtype,
            "null_count":  s.isna().sum(),
            "null_pct":    f"{s.isna().mean()*100:.1f}%",
            "unique":      s.nunique(),
            "sample":      str(s.dropna().iloc[0]) if len(s.dropna()) else "—",
            "mem_KB":      f"{mem/1024:.2f}",
        })
    return pd.DataFrame(rows)

astype() — The Core Method

astype() is pandas' primary dtype conversion tool. It is strict — it will raise on bad data by default.

astype() — all forms and options

Core API
df = raw.copy()

# ── Single column ────────────────────────────────────────────────
df["qty"]     = df["qty"].astype("int32")
df["order_id"] = df["order_id"].astype("int32")

# ── Multiple columns at once via dict ────────────────────────────
df = df.astype({
    "qty":   "int32",
    "order_id": "int32",
})

# ── Using numpy dtype objects ────────────────────────────────────
df["qty"] = df["qty"].astype(np.int16)

# ── Using Python built-in types ─────────────────────────────────
df["qty"] = df["qty"].astype(float)   # float → float64
df["qty"] = df["qty"].astype(int)     # int   → platform int (int32 or int64)

# ── errors= parameter (pandas 1.1+) ─────────────────────────────
# errors="raise"  → raise ValueError on bad data (default)
# errors="ignore" → return original column unchanged if conversion fails
df["qty"] = df["qty"].astype("int32", errors="ignore")

# ── copy= parameter ──────────────────────────────────────────────
# copy=True  → returns a new array (default, safe)
# copy=False → avoids copy when possible (optimization, be careful)
s_new = df["qty"].astype("int64", copy=False)

# ── What astype CANNOT handle gracefully ────────────────────────
try:
    df["amount"].astype(float)   # "$29.99" has "$" → ValueError!
except ValueError as e:
    print(f"astype failed: {e}")   # need to clean string first

# ── Check result ────────────────────────────────────────────────
print(df[["order_id","qty"]].dtypes)
Output
astype failed: could not convert string to float: '$29.99'

order_id    int32
qty         int32
dtype: object
astype() is strict: Use it only when you are certain the data is already clean and compatible. For raw/dirty data, use pd.to_numeric(errors="coerce"), pd.to_datetime(errors="coerce"), or manual cleaning first, then astype().

Safe cast pattern — clean then astype

# ── Pattern: pre-validate, then cast ────────────────────────────
def safe_astype(series: pd.Series, dtype, fill=np.nan):
    """Try astype; on failure coerce to fill value."""
    try:
        return series.astype(dtype)
    except (ValueError, TypeError):
        coerced = pd.to_numeric(series, errors="coerce").fillna(fill)
        return coerced.astype(dtype)

# Amount: strip "$" then cast
df["amount"] = (
    raw["amount"]
    .str.replace(r"[$,]", "", regex=True)
    .str.strip()
    .astype("float64")
)

print(df[["amount"]].dtypes)
print(df["amount"].head())

Integer overflow — picking the right int size

# Overflow is silent in NumPy — always check range before downcasting!
s = pd.Series([200, 150, 180])

print(f"min={s.min()}  max={s.max()}")

# int8 range: -128 to 127 → 180 OVERFLOWS!
print(s.astype("int8").tolist())   # ← -56, -106, -76  WRONG! silent overflow

# Safe: find the right dtype automatically
def find_int_dtype(series: pd.Series, nullable: bool = False) -> str:
    mn, mx = series.min(), series.max()
    prefix = "I" if nullable else "i"   # capital = nullable pandas type
    for bits in [8, 16, 32, 64]:
        info = np.iinfo(f"int{bits}")
        if info.min <= mn and mx <= info.max:
            return f"Int{bits}" if nullable else f"int{bits}"
    return "Int64" if nullable else "int64"

test_data = pd.Series([10, 200, 180, 90])
dtype = find_int_dtype(test_data)
print(f"\nsafe dtype: {dtype}")   # int16 — avoids int8 overflow
print(test_data.astype(dtype).tolist())  # [10, 200, 180, 90] ✅
Output
min=150  max=200
[-56, -106, -76]   ← SILENT OVERFLOW — int8 can't hold 150–200 ⚠️

safe dtype: int16
[10, 200, 180, 90]  ✅
⚠️ NumPy integer overflow is silent: Unlike Python's arbitrary-precision int, NumPy wraps around on overflow without raising an error. Always call find_int_dtype() or check s.min()/s.max() against np.iinfo() before downcasting.

Converting to Boolean

String / numeric → bool and nullable boolean

Boolean
df = raw.copy()

# ── String "True"/"False" → bool ─────────────────────────────────
# astype("bool") does NOT parse "True"/"False" strings correctly!
print(pd.Series(["True", "False"]).astype(bool).tolist())
# → [True, True]  ← "False" string is truthy! WRONG ⚠️

# ── Correct way: map strings explicitly ─────────────────────────
TRUE_VALS  = {"true","yes","1","y","on","t"}
FALSE_VALS = {"false","no","0","n","off","f"}

def parse_bool(s: pd.Series) -> pd.Series:
    lower = s.str.lower().str.strip()
    result = pd.array([pd.NA] * len(s), dtype="boolean")
    result[lower.isin(TRUE_VALS)]  = True
    result[lower.isin(FALSE_VALS)] = False
    return pd.Series(result)

df["is_paid"] = parse_bool(df["is_paid"])
print(df[["is_paid"]].dtypes)
print(df["is_paid"].tolist())

# ── Numeric → bool ──────────────────────────────────────────────
# Any non-zero int is True, 0 is False
flags = pd.Series([0, 1, 0, 1, 1])
print(flags.astype(bool).tolist())   # [False, True, False, True, True] ✅

# ── pandas nullable boolean (BooleanDtype) ───────────────────────
# Holds pd.NA for missing; regular bool casts NA to True (NaN is truthy)
s_bool_na = pd.array([True, False, pd.NA, True], dtype="boolean")
print(s_bool_na)

s_regular = pd.array([True, False, pd.NA, True], dtype="boolean")
print(f"logical AND with NA: {s_regular[0] & s_regular[2]}")  # True & NA = NA
print(f"logical OR  with NA: {s_regular[1] | s_regular[2]}")  # False | NA = NA
Output
[True, True]   ← WRONG! "False" string is truthy in Python ⚠️

After parse_bool:
is_paid    boolean

[True, False, True, True, False]  ✅

BooleanDtype with NA:
[True, False, <NA>, True]

logical AND with NA: <NA>    ← three-valued logic (SQL-style)
logical OR  with NA: <NA>
⚠️ Never use astype(bool) on string columns: Any non-empty string is truthy in Python, so "False".astype(bool) == True. Always map string representations explicitly.

Categorical Dtype

Convert low-cardinality string columns to category for dramatic memory savings and faster groupby/sort operations.

object → category — when and how

Categorical
df = raw.copy()

# ── Simple conversion ────────────────────────────────────────────
df["status"] = df["status"].astype("category")
print(df["status"].dtype)        # category
print(df["status"].cat.categories)  # Index(['cancelled', 'pending', 'shipped'])
print(df["status"].cat.codes)      # integer codes (0=cancelled, 1=pending, 2=shipped)

# ── CategoricalDtype with custom ordered categories ──────────────
from pandas.api.types import CategoricalDtype

priority_dtype = CategoricalDtype(
    categories=["pending", "shipped", "cancelled"],
    ordered=True   # enables < > comparisons
)
df["status_ord"] = df["status"].astype(priority_dtype)

# Ordered comparison — only works with ordered categories
print(df[df["status_ord"] <= "shipped"]["status_ord"].tolist())
# → ['shipped', 'pending', ...]

# ── Memory comparison ────────────────────────────────────────────
N = 1_000_000
obj_col = pd.Series(np.random.choice(["shipped","pending","cancelled"], N))
cat_col = obj_col.astype("category")

obj_mem = obj_col.memory_usage(deep=True) / 1024**2
cat_mem = cat_col.memory_usage(deep=True) / 1024**2
print(f"\n1M rows:")
print(f"  object   : {obj_mem:.2f} MB")
print(f"  category : {cat_mem:.2f} MB")
print(f"  reduction: {(1 - cat_mem/obj_mem)*100:.0f}%")

# ── When NOT to use category ─────────────────────────────────────
# category adds overhead when cardinality is HIGH
def should_categorise(s: pd.Series) -> bool:
    """Recommend category if unique ratio < 10% of rows."""
    return s.nunique() / len(s) < 0.1

print(f'\nshould_categorise(status)   : {should_categorise(df["status"])}')   # True
print(f'should_categorise(order_id) : {should_categorise(df["order_id"])}')  # False

Nullable Integer & String Types

NumPy integers cannot hold NaN. pandas 1.0+ introduced nullable versions (Int8–Int64, string, boolean) that use pd.NA instead of forcing float conversion.

The NaN-in-integer problem and the fix

Nullable
# ── The old problem: int column with NaN forces float ────────────
s_with_nan = pd.Series([1, 2, np.nan, 4])
print(f"dtype: {s_with_nan.dtype}")     # float64  ← forced to float by NaN
print(s_with_nan.tolist())            # [1.0, 2.0, nan, 4.0]  ← ugly floats

# ── The fix: pandas nullable integer dtype ───────────────────────
s_nullable = pd.array([1, 2, pd.NA, 4], dtype="Int64")
print(f"dtype: {s_nullable.dtype}")    # Int64  ← capital I = nullable
print(s_nullable)                     # [1, 2, <NA>, 4]  ← clean integers

# ── Convert a float column (born with NaN) back to nullable int ──
df = raw.copy()
df["rating"] = pd.to_numeric(df["rating"], errors="coerce")   # float64
print(f"After to_numeric: {df['rating'].dtype}")              # float64 (NaN present)

print(f"After astype Int8 : {df['rating'].dtype}")             # Int8
print(df["rating"].tolist())                                    # [4, 5, <NA>, 3, 5]

Production Dtype Pipeline

Complete dtype conversion pipeline — schema-driven

Production
🏭 Pattern: Define a schema dict once. Apply it to every raw DataFrame loaded from CSV/API. A single function converts and validates all columns.
from dataclasses import dataclass, field
from typing import Any, Optional

# ── Schema definition ────────────────────────────────────────────
ORDER_SCHEMA = {
    "order_id":   {"dtype": "int32",        "required": True},
    "date":       {"dtype": "datetime",     "format": "%Y-%m-%d"},
    "amount":     {"dtype": "float64",     "clean": lambda s: s.str.replace(r"[$,]","",regex=True)},
    "qty":        {"dtype": "int16"},
    "is_paid":    {"dtype": "boolean",     "true": ["true","yes","1"], "false": ["false","no","0"]},
    "status":     {"dtype": "category",    "categories": ["pending","shipped","cancelled"]},
    "rating":     {"dtype": "Int8"},        # nullable int
    "duration_s": {"dtype": "timedelta",   "unit": "s"},
}

def apply_schema(df: pd.DataFrame, schema: dict) -> pd.DataFrame:
    """Apply dtype schema to a raw DataFrame."""
    df = df.copy()
    errors = []

    for col, spec in schema.items():
        if col not in df.columns:
            if spec.get("required"):
                errors.append(f"Required column '{col}' missing")
            continue

        s = df[col]

        # Pre-clean hook
        if "clean" in spec:
            s = spec["clean"](s)

        dtype = spec["dtype"]
        try:
            if dtype == "datetime":
                fmt = spec.get("format")
                s = pd.to_datetime(s, format=fmt, errors="coerce")
            elif dtype == "timedelta":
                s = pd.to_timedelta(pd.to_numeric(s, errors="coerce"), unit=spec.get("unit","s"))
            elif dtype == "boolean":
                lower = s.astype(str).str.lower().str.strip()
                t, f  = spec.get("true",["true"]), spec.get("false",["false"])
                arr   = pd.array([pd.NA]*len(s), dtype="boolean")
                arr[lower.isin(t)] = True
                arr[lower.isin(f)] = False
                s = pd.Series(arr, index=df.index)
            elif dtype == "category":
                cats = spec.get("categories")
                s = s.astype("category")
                if cats: s = s.cat.set_categories(cats)
            else:
                s = pd.to_numeric(s, errors="coerce").astype(dtype)
            df[col] = s
        except Exception as e:
            errors.append(f"Column '{col}': {e}")

    if errors:
        print("⚠️ Schema errors:")
        for err in errors: print(f"  • {err}")

    return df

# ── Apply it ─────────────────────────────────────────────────────
df_typed = apply_schema(raw, ORDER_SCHEMA)

print("\nDtypes after schema:")
print(df_typed.dtypes)
print(f"\nMemory before: {raw.memory_usage(deep=True).sum()/1024:.1f} KB")
print(f"Memory after : {df_typed.memory_usage(deep=True).sum()/1024:.1f} KB")

Column type → recommended dtype decision tree

What kind of data is in the column?
│
├─ Whole numbers (no NaN) ──────────────────────────────────────────
│   └─ What is the value range?
│       ├─ -128..127            → int8
│       ├─ -32768..32767        → int16
│       ├─ ±2.1 billion         → int32
│       └─ larger               → int64
│
├─ Whole numbers (WITH NaN) ────────────────────────────────────────
│   └─ Use nullable type: Int8 / Int16 / Int32 / Int64
│       (never float64 just to hold NaN)
│
├─ Decimal numbers ─────────────────────────────────────────────────
│   ├─ Precision needed (finance, science) → float64
│   └─ ML feature / sensor / approximate  → float32 (50% memory)
│
├─ Text / strings ──────────────────────────────────────────────────
│   ├─ Low cardinality (<10% unique)  → category  ← fastest, smallest
│   ├─ High cardinality, NaN present  → string (pd.StringDtype)
│   └─ Legacy / library compatibility → object
│
├─ True/False ──────────────────────────────────────────────────────
│   ├─ No NaN                         → bool
│   └─ NaN possible                  → boolean (pd.BooleanDtype)
│
├─ Dates & times ───────────────────────────────────────────────────
│   ├─ Date/time values               → datetime64[us] (pandas 2.x)
│   ├─ Date/time + timezone           → datetime64[us, UTC]
│   └─ Duration / elapsed time        → timedelta64
│
└─ Reporting periods (monthly, quarterly) → period["M"] / period["Q"]
⚠️ Never astype(bool) on strings:
Any non-empty string is truthy in Python. "False".astype(bool) == True. Always map string boolean representations with a dict or explicit set membership.
⚠️ Integer overflow is silent:
NumPy wraps around without raising. np.int8(200) == -56. Always check s.min()/s.max() against np.iinfo() before downcasting.
💡 astype() is strict:
Use pd.to_numeric(errors="coerce") and pd.to_datetime(errors="coerce") for raw/dirty data. Use astype() only after cleaning — it raises on any incompatible value.
✅ Always check memory_usage(deep=True):
df.memory_usage(deep=False) underreports object columns (shows pointer size, not string size). Always use deep=True for accurate measurement.
💡 category cardinality rule:
Convert to category only when nunique() / len(df) < 0.10. High-cardinality categories use more memory than plain object due to category overhead.
💡 Int8 vs int8:
Lowercase int8 is a NumPy type — no NaN support. Capital Int8 is a pandas ExtensionArray — holds pd.NA. Use capital-I types whenever the column can have missing values.
